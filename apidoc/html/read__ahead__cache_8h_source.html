113struct PAIMON_EXPORT ByteRange {
114 uint64_t offset;
115 uint64_t length;
116
117 ByteRange() = default;
118 ByteRange(uint64_t offset, uint64_t length) : offset(offset), length(length) {}
119
120 friend bool operator==(const ByteRange& left, const ByteRange& right) {
121 return (left.offset == right.offset && left.length == right.length);
122 }
123 friend bool operator!=(const ByteRange& left, const ByteRange& right) {
124 return !(left == right);
125 }
126
129 bool Contains(const ByteRange& other) const {
130 return (offset <= other.offset && offset + length >= other.offset + other.length);
131 }
132};
